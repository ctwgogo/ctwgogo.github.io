java-Character相关方法

发表于 2017-10-09 | 分类于 java | 阅读次数:
  1. 代码点：是指一个编码表中的某个字符对应的代码值，也就是Unicode编码表中每个字符对应的数值。Unicode标准中，代码点采用16进制书写，并加上前缀U+，比如字符A对于的编码值是U+0041，Unicode的代码点可以分成17个代码级别。第一个代码级别称为基本的多语言级别，代码点从U+0000到U+FFFF，其中包括了经典的Unicode代码，其余的16个附加级别，代码点从U+10000到U+10FFFF，其中包括了一些增补字符。在UTF-16中的基本多语言级别中，每一个字符用16位表示，通常被称为代码单元。因为UTF-16编码采用不同长度的编码表示所有的Unicode代码点，因此增补字符采用的是一对连续的代码单元进行编码。
  2. java String方法：
    public class TestString {
public static void main(String[] args)
{
System.out.println("字符串方法测试");
char[] str = {'d','g','e','y','\0'};
String string = new String("ctwgogo.github.io");
//1.charAt(int index):返回指定索引处的char值
System.out.println(string.charAt(2));
//2.int compareTo(Object o):把这个字符串与另外一个对象比较.
System.out.println(string.compareTo("ctw"));
//3.int compareTo(String another string):按字典顺序,比较两个字符串
System.out.println(string.compareTo("de"));
//4.int compareToIgnoreCase(String str):按字典顺序比较字符串，不考虑大小写
System.out.println(string.compareToIgnoreCase("ctw"));
//5.String concat(String str):将指定字符串连接到此字符串的结尾
System.out.println(string.concat("hehe"));
//6.boolean contentEquals(StringBuffer sb):当且仅当字符串与指定的StringButter有相同顺序的字符时候返回真
StringBuffer sb = new StringBuffer("ctwgogo");
System.out.println(string.contentEquals(sb));
//7.static String copyValueof(char[] data):返回指定数组中表示该字符序列的 String.
System.out.println(String.copyValueOf(str));
//8.boolean endsWith(String suffix):测试此字符是否以特定的字符结尾
System.out.println(string.endsWith("hehe"));
//9.boolean equals():将字符串与指定的对象比较
System.out.println(string.equals(str));
//10.boolean equalsIgnoreCase(String anotherString):将该字符串与另外一个比较,不考虑大小写.
System.out.println(string.equalsIgnoreCase("str"));
//11.byte[] getBytes():使用平台的默认字符集将此 String 编码为 byte 序列，并将结果存储到一个新的 byte 数组中
System.out.println(string.getBytes());
//12.void getChars(int srcBegin,int srcEnd,char[] dst,int dstBegin):将字符从此字符串复制到目标字符数组.
char[] str1 = {'c','t','w','t','e','\0'};
string.getChars(3,5,str1,2);
System.out.println(str1);
//13.int hashCode():返回字符串的hash码
System.out.println(string.hashCode());
//14.int indexof(int ch):返回指定字符在此字符串中第一次出现处的索引.
System.out.println(string.indexOf('g'));
//15.int indexof(String str , int fromIndex):从fromIndex开始，找到第一个出现的字符的索引位置
System.out.println(string.indexOf('g',6));
//16.String intern():返回字符串打标准形式
System.out.println(string.intern());
//17.int lastIndexOf(int ch):返回指定字符串最后出现的地方
System.out.println(string.lastIndexOf('h'));
//18.int lastIndexOf(String str,int fromindex):
System.out.println(string.lastIndexOf('g',7));
//19.int length():返回字符串的长度
System.out.println(string.length());
 //20.boolean matches(String regex):告知此字符串是否匹配给定的正则表达式．
System.out.println(string.matches("ctw(.*)"));
//21.boolean regionMatches(boolean ignoreCase,int toffset,String other,int ooffset,int len)
//用toffset和ooffset分别指出当前字符串中的子串起始位置和要与之比较的字符串中的子串起始地址；len 指出比较长度
System.out.println(string.regionMatches(false,1,"ctw",1,2));
//22.String replace(char oldChar,char newChar):返回一个新字符串,它是通过newChar替换此字符中出现打所有oldchar得到
System.out.println(string.replace("ctw","hehe"));
//23.String replaceAll(String regex,String replacement):使用给定的replacement替换此字符串所有匹配给定的正则表达式的字串
System.out.println(string.replaceAll("ctw(.*)","ht"));
//24.String replaceFirst(String regex,String replacement):使用给定的 replacement 替换此字符串匹配给定的正则表达式的第一个子字符串
System.out.println(string.replaceFirst("ctw+b","hehe"));
//25.boolean startsWith(String prefix):测试此字符串是否以指定的前缀开始
System.out.println(string.startsWith("c"));
//26.boolean startsWith(String prefix,int toffset):测试此字符串从指定字符串开始的子字符串是否以指定前缀开始
System.out.println(string.startsWith("c",3));
//27.CharSequence subSequence(int beginIndex,int endIndex):返回一个新的字符序列，它是此序列的一个子字符串
System.out.println(string.subSequence(1,3));
//28.String substring(int beginIndex,int endIndex):返回一个新的字符串,它是此字符串打一个子字符串
System.out.println(string.substring(4,6));
//29.char[] toCharArray():将此字符串转换成为一个新的字符数组
char[] str2 = string.toCharArray();
System.out.print(str2);
//30.String toLowerCase():所有字符变成小写(默认语言环境)
System.out.println("HeHeHe".toLowerCase());
//31.String toUpperCase():所有字符变成大写
System.out.println("hehehe".toUpperCase());
//32.String trim():返回字符串的副本，忽略前导空白和尾部空白
System.out.println("   heh0".trim());
    }
}
